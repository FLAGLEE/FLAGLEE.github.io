static void swap(void* a, void* b, size_t size)
{
    char* p1 = (char*)a;
    char* p2 = (char*)b;
    char temp;
    while (size--)
    {
        temp = *p1;
        *p1 = *p2;
        *p2 = temp;
        p1++;
        p2++;
    }
}

static int PartSort(double* arr, int start, int end)
{
    int left = start;
    int right = end;
    double key = arr[end];   //é€‰å–å…³é”®å­—
    while (left < right)
    {
        while (left < right && arr[left] <= key)  //å·¦è¾¹æ‰¾æ¯”keyå¤§çš„å€¼

        }

/* ave */
void cal_ave(double rr_temp[][3], int size, double ave_rr[]) {
    double sum[3] = { 0 };
    int i, j;

    for (i = 0; i < size; i++) {
        for (j = 0; j < 3; j++) {
            sum[j] += rr_temp[i][j];
        }
    }

    for (i = 0; i < 3; i++) {
        ave_rr[i] = sum[i] / size;
    }

}


/* std */
void cal_std(double rr_temp[][3], int size, double ave_rr[], double std_rr[]) {
    double sum[3] = { 0 };
    int i, j;

    for (i = 0; i < size; i++) {
        for (j = 0; j < 3; j++) {
            sum[j] += (rr_temp[i][j] - ave_rr[j]) * (rr_temp[i][j] - ave_rr[j]);
        }
    }

    for (i = 0; i < 3; i++) {
        std_rr[i] = sqrt(sum[i] / size);
    }
}
