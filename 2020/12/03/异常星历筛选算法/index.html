异常星历筛选算法

FLAG

GNSS

发布于：Dec 3, 2020

近期在各种项目中，总是不可避免地遇到星历异常问题，而广播星历的误差会对 GPS 定位精度会产生严重影响。因此，设计一种监测数据实时解算星历异常处理方法极为重要。

这篇文章记录一下做的第二个工作，剔除星历文件中异常的卫星，得到最优星历。

异常处理方法

当接收到融合后的到广播星历实时融合流和到广播星历实时接收流之后，首先要验证其中是否有错误的卫星星历，考虑到变形监测的实时性需要，我们采用求解 SPP 的办法来简单高效地验证星历异常问题。验证流程如下图所示，因为实际应用中多系统 GNSS 星历异常检验过于复杂，这里我们采用 GPS+BDS 双系统做简要讲解。

首先，采用原始的接收到的 GPS+BDS 双系统的星历去计算 SPP，若有解，判断解的历元数是否大于观测数据历元数的 1/2 及解的标准偏差是否小于 1000（这里的 1/2 和 1000 为认为设置的阈值，目的是排除星历错误但仍有解、部分时段星历错误但仍有解等其他情况），若满足则输出解，若不满足则与双系统无解一样去分别计算单系统情况。

若 BDS 有解，则接下来尝试逐次移除 GPS 卫星组合来尝试双系统能否有解。举例说明，若剔除 G01、G02 后 SPP 解算正常也能通过阈值检验，那么便可以输出有问题的卫星和解算结果。单 GPS 有解也是类似操作，但由于 GPS 卫星分轨道的特殊性，BDS 的地球静止轨道卫星（GEO）和倾斜地球同步轨道卫星（IGSO）存在较频繁的轨道机动[8]，因此多了一步剔除某个轨道卫星的操作。

当我们得到完整解或提出错误卫星的解之后，在比较两种星历解的精度指标，比如解算卫星数、结果标准差等指标，每个指标赋予相应权重，挑选出最优的星历，用于实时变形监测工程。

static void swap(void* a, void* b, size_t size)
{
    char* p1 = (char*)a;
    char* p2 = (char*)b;
    char temp;
    while (size--)
    {
        temp = *p1;
        *p1 = *p2;
        *p2 = temp;
        p1++;
        p2++;
    }
}

static int PartSort(double* arr, int start, int end)
{
    int left = start;
    int right = end;
    double key = arr[end];   //选取关键字
    while (left < right)
    {
        while (left < right && arr[left] <= key)  //左边找比key大的值
        {
            ++left;
        }
        while (left < right && arr[right] >= key)  //右边找比key小的值
        {
            --right;
        }
        if (left < right)
        {
            swap(&arr[left], &arr[right], sizeof(double));  //找到之后交换左右的值
        }
    }
    swap(&arr[right], &arr[end], sizeof(double));
    return left;
}
extern int getmedium_array(const double* arr, const int size, double* median)
{
    int start = 0;
    int end = size - 1;
    int mid = (size - 1) / 2;
    int div = PartSort(arr, start, end);
    while (div != mid)
    {
        if (mid < div) //左半区间找
            div = PartSort(arr, start, div - 1);
        else   //左半区间找
            div = PartSort(arr, div + 1, end);
    }
    if (median)* median = arr[mid];
    return 0;
}

/* ave */
void cal_ave(double rr_temp[][3], int size, double ave_rr[]) {
    double sum[3] = { 0 };
    int i, j;

    for (i = 0; i < size; i++) {
        for (j = 0; j < 3; j++) {
            sum[j] += rr_temp[i][j];
        }
    }

    for (i = 0; i < 3; i++) {
        ave_rr[i] = sum[i] / size;
    }

}


/* std */
void cal_std(double rr_temp[][3], int size, double ave_rr[], double std_rr[]) {
    double sum[3] = { 0 };
    int i, j;

    for (i = 0; i < size; i++) {
        for (j = 0; j < 3; j++) {
            sum[j] += (rr_temp[i][j] - ave_rr[j]) * (rr_temp[i][j] - ave_rr[j]);
        }
    }

    for (i = 0; i < 3; i++) {
        std_rr[i] = sqrt(sum[i] / size);
    }
}

/* exclude satellite */
void exclude_sat(char str_sats[], rtk_t* rtk, int print) {
    char buff[1024];
    char* sat_p, * sat_id;
    int sat_prn;
    int sat_prns[MAXSAT] = { 0 };
    int count = 0;
    char str_sat_prns[512];
    char temp_sat_prns[4];

    for (int i = 0; i < MAXSAT; i++) rtk->opt.exsats[i] = 0;
    strcpy(buff, str_sats);
    /* strtok返回被分解的第一个子字符串，如果没有可检索的字符串，则返回一个空指针;分割后的str会变 */
    for (sat_p = strtok(buff, " "); sat_p; sat_p = strtok(NULL, " ")) {
        if (*sat_p == '+') sat_id = sat_p + 1; else sat_id = sat_p;
        if (!(sat_prn = satid2no(sat_id))) continue;
        rtk->opt.exsats[sat_prn - 1] = *sat_p == '+' ? 2 : 1;
        sat_prns[count++] = sat_prn;
    }

    if (print) {
        strcpy(str_sat_prns, "\texclude sat : ");

        for (int i = 0; i < count; i++) {

            //itoa(sat_prns[i], temp_sat_prns, 10);
            sprintf(temp_sat_prns, "%d", sat_prns[i]);
            strcat(str_sat_prns, temp_sat_prns);
            strcat(str_sat_prns, " ");
            //trace(1, "%d ", sat_prns[i]);
        }
        trace(1, "%s\n", str_sat_prns);
        //trace(1, "\n");
    }
}

/* get satellite group */
int get_sat_group(char str_sats[], int sat_prns[]) {
    char buff[1024];
    char* sat_p, * sat_id;
    int sat_prn, count = 0;

    strcpy(buff, str_sats);
    /* strtok返回被分解的第一个子字符串，如果没有可检索的字符串，则返回一个空指针;分割后的str会变 */
    for (sat_p = strtok(buff, " "); sat_p; sat_p = strtok(NULL, " ")) {
        if (*sat_p == '+') sat_id = sat_p + 1; else sat_id = sat_p;
        if (!(sat_prn = satid2no(sat_id))) continue;
        sat_prns[count++] = sat_prn;
    }

    return count;
}
